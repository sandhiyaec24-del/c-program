#include <stdio.h>
#include <string.h>

struct Address {
    char city[100];
    int pincode;
};

struct Student {
    int id;
    char name[100];
    struct Address address;
};

void printStudentDetails(struct Student *student) {
    printf("Student ID: %d\n", student->id);
    printf("Student Name: %s\n", student->name);
    printf("City: %s\n", student->address.city);
    printf("Pincode: %d\n", student->address.pincode);
}

int main() {
    struct Student student;
    student.id = 1;
    strcpy(student.name, "John Doe");
    strcpy(student.address.city, "New York");
    student.address.pincode = 10001;

    printf("Student Details:\n");
    printStudentDetails(&student);

    return 0;
}


Problem 2: Employee Department Management System
#include <stdio.h>
#include <string.h>

struct Department {
    char name[100];
    int id;
};

struct Employee {
    int id;
    char name[100];
    struct Department department;
};

void printEmployeeDetails(struct Employee *employee) {
    printf("Employee ID: %d\n", employee->id);
    printf("Employee Name: %s\n", employee->name);
    printf("Department Name: %s\n", employee->department.name);
    printf("Department ID: %d\n", employee->department.id);
}

int main() {
    struct Employee employee;
    employee.id = 1;
    strcpy(employee.name, "Jane Doe");
    strcpy(employee.department.name, "HR");
    employee.department.id = 101;

    printf("Employee Details:\n");
    printEmployeeDetails(&employee);

    return 0;
}
Problem 3: Product Supplier Management System

#include <stdio.h>
#include <string.h>
#include <stdlib.h>

struct Supplier {
    char name[100];
    char location[100];
};

struct Product {
    int id;
    char name[100];
    float price;
    struct Supplier supplier;
};

void printProductDetails(struct Product *product) {
    printf("Product ID: %d\n", product->id);
    printf("Product Name: %s\n", product->name);
    printf("Product Price: %.2f\n", product->price);
    printf("Supplier Name: %s\n", product->supplier.name);
    printf("Supplier Location: %s\n", product->supplier.location);
}

int main() {
    struct Product *product = (struct Product *)malloc(sizeof(struct Product));
    product->id = 1;
    strcpy(product->name, "Laptop");
    product->price = 999.99;
    strcpy(product->supplier.name, "ABC Inc.");
    strcpy(product->supplier.location, "New York");

    printf("Product Details:\n");
    printProductDetails(product);

    free(product);

    return 0;
}



#include<stdio.h>
#include<stdlib.h>
struct callnumber{
    int number;
    struct callnumber *next;
};
int main(){
    struct callnumber *num1 = (struct callnumber*)malloc(sizeof(struct callnumber));
    struct callnumber *num2 = (struct callnumber*)malloc(sizeof(struct callnumber));
    struct callnumber *num3 = (struct callnumber*)malloc(sizeof(struct callnumber));
    scanf("%d %d %d",&num1->number,&num2->number,&num3->number);
    num1->next = num2;
    num2->next = num3;
    num3->next = NULL;
    
    struct callnumber *head = num1;
    while(head!=NULL){
    printf("%d->",head->number);
    head = head->next;
    }
    printf("NULL");
    return 0;
}
